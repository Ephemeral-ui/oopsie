package package1;
public class PersonalDetails {
private String name,address,dob,gender;
private int age;
public PersonalDetails(String name1, String address1, String dob1, String gender1, int age1) {
name=name1;
address=address1;
dob=dob1;
gender=gender1;
age=age1;
}
public void displayPersonalDetails() {
System.out.println("Student's Personal Details:");
System.out.println("Name: " + name);
System.out.println("Address: " + address);
System.out.println("Birth date: " + dob);
System.out.println("gender: " + gender);
System.out.println("Age: " + age);
}
}
