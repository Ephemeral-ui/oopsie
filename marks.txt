import java.util.Scanner;

public class ExamResults {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        // Constants
        final int NUM_STUDENTS = 5;
        final int NUM_SUBJECTS = 3;

        // Array to store roll numbers and marks
        int[][] marks = new int[NUM_STUDENTS][NUM_SUBJECTS + 1]; // One extra for roll number

        // Read marks for each student
        for (int i = 0; i < NUM_STUDENTS; i++) {
            System.out.println("Enter Roll Number for Student " + (i + 1) + ":");
            marks[i][0] = scanner.nextInt(); // Read Roll Number

            // Read marks for each subject
            System.out.println("Enter marks for Student " + (i + 1) + ":");
            for (int j = 1; j <= NUM_SUBJECTS; j++) {
                System.out.println("Enter marks for Subject " + j + ":");
                marks[i][j] = scanner.nextInt();
            }
        }

        // (a) Total marks obtained by each student
        System.out.println("Total marks obtained by each student:");
        for (int i = 0; i < NUM_STUDENTS; i++) {
            int totalMarks = 0;
            for (int j = 1; j <= NUM_SUBJECTS; j++) {
                totalMarks += marks[i][j];
            }
            System.out.println("Roll Number " + marks[i][0] + ": " + totalMarks);
        }

        // (b) The highest marks in each subject and the Roll No. of the student who secured it
        System.out.println("Highest marks in each subject:");
        for (int j = 1; j <= NUM_SUBJECTS; j++) {
            int maxMarks = 0;
            int rollNo = -1;
            for (int i = 0; i < NUM_STUDENTS; i++) {
                if (marks[i][j] > maxMarks) {
                    maxMarks = marks[i][j];
                    rollNo = marks[i][0];
                }
            }
            System.out.println("Subject " + j + ": " + maxMarks + " (Roll Number " + rollNo + ")");
        }

        // (c) The student who obtained the highest total marks
        int highestTotalMarks = 0;
        int highestTotalMarksRollNo = -1;
        for (int i = 0; i < NUM_STUDENTS; i++) {
            int totalMarks = 0;
            for (int j = 1; j <= NUM_SUBJECTS; j++) {
                totalMarks += marks[i][j];
            }
            if (totalMarks > highestTotalMarks) {
                highestTotalMarks = totalMarks;
                highestTotalMarksRollNo = marks[i][0];
            }
        }
        System.out.println("Student with the highest total marks: Roll Number " + highestTotalMarksRollNo);
    }
}
